218 double find_reasonable_epsilon(double eps,
219  ADExprType& ad_expr,
220  MatType& theta,
221  MatType& theta_adj,
222  const MomentumHandlerType& momentum_handler)
223 {
224  // See (STAN) for reference: if epsilon is way out of bounds, just return eps
225  if (eps == 0 || eps > 1e7) return eps;
226 
227  const double diff_bound = std::log(0.8);
228 
229  arma::mat::fixed<n_params, 1> r_mat(arma::fill::zeros);
230  auto r = r_mat.col(0);
231 
232  arma::mat::fixed<n_params, 2> theta_mat(arma::fill::zeros);
233  auto theta_orig = theta_mat.col(0);
234  auto theta_adj_orig = theta_mat.col(1);
235 
236  // sample momentum vector based on handler
237  momentum_handler.sample(r);
238 
239  // differentiate first to get adjoints and hamiltonian
240  const double potential_orig = -ad::autodiff(ad_expr);
241  double kinetic_orig = momentum_handler.kinetic(r);
242  double ham_orig = hamiltonian(potential_orig, kinetic_orig);
243 
244  // save original value and adjoint
245  theta_orig = theta;
246  theta_adj_orig = theta_adj;
247 
248  // get current hamiltonian after leapfrog
249  double potential_curr = leapfrog(
250  ad_expr, theta, theta_adj, r, momentum_handler, eps, true);
251  double kinetic_curr = momentum_handler.kinetic(r);
252  double ham_curr = hamiltonian(potential_curr, kinetic_curr);
253 
254  int a = (ham_orig - ham_curr > diff_bound) ? 1 : -1;
255 
256  while (1) {
257 
258  // check if break condition holds
259  if ( ((a == 1) && !(ham_orig - ham_curr > diff_bound)) ||
260  ((a == -1) && !(ham_orig - ham_curr < diff_bound)) ) {
261  break;
262  }
263 
264  // update epsilon
265  eps *= (a == -1) ? 0.5 : 2;
266 
267  // copy back original value and adjoint
268  theta = theta_orig;
269  theta_adj = theta_adj_orig;
270 
271  // recompute original hamiltonian with new momentum
272  momentum_handler.sample(r);
273  kinetic_orig = momentum_handler.kinetic(r);
274  ham_orig = hamiltonian(potential_orig, kinetic_orig);
275 
276  // leapfrog and compute current hamiltonian
277  potential_curr = leapfrog(
278  ad_expr, theta, theta_adj, r, momentum_handler, eps, true);
279  kinetic_curr = momentum_handler.kinetic(r);
280  ham_curr = hamiltonian(potential_curr, kinetic_curr);
281 
282  }
283 
284  // copy back original value and adjoint
285  theta = theta_orig;
286  theta_adj = theta_adj_orig;
287 
288  return eps;
289 }
